#include <stdio.h>
#include <stdlib.h>

unsigned int func1(int a1)

    unsigned int v2 = 0;

        v2 += 3;

}

int main(int argc, char **argv)
{
    int v3 = atoi(argv[1]);
    unsigned int v4 = func1(v3);
    return printf("Result: %ld\n", v4);
}
